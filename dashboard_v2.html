الصق الكود التالي كاملاً

احذف الكود الموجود والصق هذا مكانه تماماً:

function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheets()[0]; var data = sheet.getDataRange().getValues(); var rows = []; var start = (data[0] && (String(data[0][0]).includes('اريخ') || String(data[0][0]).toLowerCase().includes('date'))) ? 1 : 0; for (var i = start; i < data.length; i++) { var r = data[i]; if (!r[0] && !r[1]) continue; rows.push({date:String(r[0]||''),name:String(r[1]||''),type:String(r[2]||''),device:String(r[3]||''),reason:String(r[4]||''),next:String(r[5]||'')}); } var out = ContentService.createTextOutput(JSON.stringify({rows:rows,total:rows.length})); out.setMimeType(ContentService.MimeType.JSON); return out; } function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheets()[0]; if (sheet.getLastRow() === 0) sheet.appendRow(['التاريخ','اسم المراجعة','النوع','الجهاز','سبب الإلغاء','الموعد القادم']); var d = JSON.parse(e.postData.contents); sheet.appendRow([d.date||'',d.name||'',d.type||'',d.device||'',d.reason||'',d.next||'-']); var out = ContentService.createTextOutput(JSON.stringify({success:true})); out.setMimeType(ContentService.MimeType.JSON); return out; }
